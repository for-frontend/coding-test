# 1. 내 풀이
- 이 문제는 풀기 전에 스터디 때 다른 분들이 푸신 과정을 들었어서, 그 프로세스를 기억하며 풀어보았습니다.

```javascript
function solution(r1, r2) {
    var answer = 0;

    for (let i = 1; i <= r2; i++) {
        let max = Math.floor(Math.sqrt(r2**2 - i**2));
        let min = i >= r1
            ? 0
            : Math.ceil(Math.sqrt(r1**2 - i**2));

        answer = answer + max - min + 1;
    }
    return answer * 4;
}
```
